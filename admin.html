كيفية استخدام لوحة التحكم:

  1. انقر على "إضافة إعلان جديد" لإضافة عقار جديد

  3. انقر على "الطلبات الواردة" لمشاهدة طلبات العملاء

دخول إلى لوحة التحكم // عند تحميل الصفحة، تهيئة البيانات initializeData(); // معالجة إضافة إعلان جديد document.getElementById('addPropertyForm').addEventListener('submit', function(e) { e.preventDefault(); const property = { title: document.getElementById('property-title').value, type: document.getElementById('property-type').value, operation: document.getElementById('operation-type').value, price: document.getElementById('price').value, location: document.getElementById('location').value, description: document.getElementById('description').value }; addProperty(property); alert('تم إضافة الإعلان بنجاح!'); this.reset(); // تحديث قائمة العقارات refreshPropertiesList(); }); // تحديث قائمة العقارات function refreshPropertiesList() { const properties = getProperties(); const tbody = document.querySelector('#manage-properties table tbody'); if (properties.length === 0) { tbody.innerHTML = 'لا توجد إعلانات'; return; } tbody.innerHTML = properties.map(property => ` ${property.title} ${property.type} ${property.operation === 'بيع' ? property.price + ' دج' : property.price + ' دج/شهر'} ${property.location} `).join(''); } // تحديث قائمة الطلبات function refreshRequestsList() { const requests = getRequests(); const tbody = document.querySelector('#view-requests table tbody'); if (requests.length === 0) { tbody.innerHTML = 'لا توجد طلبات'; return; } tbody.innerHTML = requests.map(request => ` ${request.name || 'غير محدد'} ${request.phone || 'غير محدد'} ${request.service_type || 'غير محدد'} ${request.date || 'غير محدد'} ${request.status || 'جديد'} `).join(''); } // عند تحميل الصفحة، عرض البيانات document.addEventListener('DOMContentLoaded', function() { refreshPropertiesList(); refreshRequestsList(); });
